%\newcommand{\sflegElem}[2][1em]{%

\newcommand{\sflegElem}[2][1.2em]{%
	\begin{tikzpicture}
		\figfont
    %\node[fill=col_gray_light!30, minimum height=#1, minimum width=#1, inner sep=0pt, draw]{\strut#2};
    \node[fill=col_gray_light, text=black, minimum height=#1, minimum width=#1, inner sep=2pt]{(#2)};
\end{tikzpicture}%
}

\newcommand{\sflegTwo}[1][0ex]{
\begin{minipage}{0.2\textwidth}
	\vspace{#1}
	\sflegElem{a}\,\sflegElem{b}
\end{minipage}
\vspace{-2ex}
}
\newcommand{\sflegOneTwo}[1][0ex]{
	\begin{flushleft}
	\vspace{#1}
	\sflegElem[2.75em]{a}\\[0.75pt]
	\sflegElem{b}\,\sflegElem{c}
	\end{flushleft}
}
\newcommand{\sflegTwoTwo}[1][0ex]{
\begin{minipage}{0.2\textwidth}
	\vspace{#1}
	\sflegElem{a}\,\sflegElem{b}\\[0.75pt]
	\sflegElem{c}\,\sflegElem{d}
\end{minipage}
\vspace{-2ex}
}
